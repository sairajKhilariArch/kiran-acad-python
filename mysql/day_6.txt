			-- inner join
            
select s1.sname , c1.cname from 
std_info as s1 inner join course_info as c1
on s1.sid = c1.sid;

-- natural join

select * from 
std_info s1 natural join course_info c1;

-- left outer join

select s1.*, c1.cname from 
std_info s1 left join course_info c1
on s1.sid = c1.sid;

-- right outer join

select s1.sid, s1.sname, c1.* from
std_info s1 right outer join course_info c1
on s1.sid = c1.sid ;

-- full outer Join

select s1.*, c1.cname from 
std_info s1 left join course_info c1
on s1.sid = c1.sid
union
select s1.sid, s1.sname, c1.* from
std_info s1 right outer join course_info c1
on s1.sid = c1.sid ;

-- cross join 

select * from std_info cross join course_info;


-- self join 

select * from 
std_info s1 join  std_info s2
on s1.sid = s2.sid;
